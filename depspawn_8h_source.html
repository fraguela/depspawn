482  template<typename T_it, typename Head, typename... Tail>
483  int fill_args(arg_info*& args, Head&& h, Tail&&... t) {
484  typedef typename boost::mpl::deref<T_it>::type curr_t; //Formal parameter type
485  constexpr bool is_reference = std::is_reference<curr_t>::value;
486  typedef typename std::remove_reference<curr_t>::type deref_t;
487  constexpr bool is_const = std::is_const<deref_t>::value;
488  constexpr bool is_writable = is_reference && !is_const;
489  constexpr bool is_barray = is_blitz_array<typename std::remove_const<typename std::remove_reference<Head>::type>::type>::value;
490  constexpr int process_argument =
491  !is_ignored<curr_t&&>::value &&
492  !is_ignored<Head&&>::value &&
493  !(std::is_rvalue_reference<Head&&>::value && !is_barray);
494 
495  if(process_argument) {
496 
497  arg_info* n = arg_info::Pool.malloc();
498  n->size = sizeof(Head);
499  n->wr = is_writable;
500  n->next = nullptr;
501  n->fill_in_array(h);
502 
503  // Insert new_arg (n) in order
504  n->insert_in_arglist(args);
505  }
506 
507  return process_argument + fill_args<typename boost::mpl::next<T_it>::type>(args, std::forward<Tail>(t)...);
508  }

708  lbs2 = r2.last() -j +1 <= bs2 ? r2.last() -j +1 : bs2;
